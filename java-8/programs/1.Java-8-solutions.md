## 1. Find the Frequency of Each Character in a Given String

```java
import java.util.Map;
import java.util.function.Function;
import java.util.stream.Collectors;

public class CharacterFrequency {
    public static void main(String[] args) {
        String input = "programming";

        // chars() → IntStream; groupingBy groups by char; counting() tallies each group
        Map<Character, Long> frequencyMap = input.chars()
                .mapToObj(c -> (char) c)
                .collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));

        frequencyMap.forEach((character, count) ->
            System.out.println(character + ": " + count));
    }
}
```


### Approach 2 - Using Traditional Loop with Stream Enhancement

```java
Map<Character, Integer> freqMap = new HashMap<>();

// forEach + merge
input.chars().forEach(c -> freqMap.merge((char) c, 1, Integer::sum));

// getOrDefault alternative
input.chars().forEach(c -> freqMap.put((char) c, freqMap.getOrDefault((char) c, 0) + 1));

// One-liner: merge function resolves duplicate keys by summing

        .mapToObj(c -> (char) c)
        .collect(Collectors.toMap(c -> c, c -> 1, Integer::sum));
```

## 2. Find the Maximum Number and Minimum Number in a List

```java
import java.util.Arrays;
import java.util.List;

public class MinMaxInList {
    public static void main(String[] args) {
        List<Integer> numbers = Arrays.asList(5, 9, 1, 3, 4, 8, 6, 7, 2);

        // max/min return Optional; orElseThrow guards empty list
        Integer max = numbers.stream()
                .max(Integer::compare)
                .orElseThrow(() -> new IllegalStateException("No maximum value found"));

        Integer min = numbers.stream()
                .min(Integer::compare)
                .orElseThrow(() -> new IllegalStateException("No minimum value found"));

        // mapToInt enables primitive stream's max()/min()
        int maxPrimitive = numbers.stream().mapToInt(Integer::intValue).max()
                .orElseThrow(() -> new IllegalStateException("No maximum value found"));

        int minPrimitive = numbers.stream().mapToInt(Integer::intValue).min()
                .orElseThrow(() -> new IllegalStateException("No minimum value found"));
    }
}
```

## 3. Find the Second Largest Number in the List of Integers

```java
import java.util.Arrays;
import java.util.Comparator;
import java.util.List;
import java.util.Optional;

public class SecondLargestNumber {
    public static void main(String[] args) {
        List<Integer> numbers = Arrays.asList(5, 9, 1, 3, 9, 8, 6, 7, 2);

        // distinct() removes duplicates; skip(1) skips the largest
        Optional<Integer> secondLargest = numbers.stream()
                .sorted(Comparator.reverseOrder())
                .distinct()
                .skip(1)
                .findFirst();

        // Alternative: limit top 2, then pick the smaller
        Optional<Integer> secondLargest2 = numbers.stream()
                .distinct()
                .sorted(Comparator.reverseOrder())
                .limit(2)
                .min(Comparator.naturalOrder());
    }
}
```

## 5. Find the Sum of All Digits of a Number

```java
import java.util.stream.IntStream;

public class SumOfDigits {
    public static void main(String[] args) {
        int number = 12345;

        // Convert to string; getNumericValue maps '3' → 3
        int sum1 = String.valueOf(number)
                .chars()
                .map(Character::getNumericValue)
                .sum();

        // iterate divides by 10 each step; % 10 extracts last digit
        int sum2 = IntStream.iterate(number, n -> n > 0, n -> n / 10)
                .map(n -> n % 10)
                .sum();
    }
}
```

### Approach 3 - Alternatives

```java
// Recursive
private static int sumDigitsRecursive(int number) {
    return number == 0 ? 0 : (number % 10) + sumDigitsRecursive(number / 10);
}

// c - '0' converts char digit to int
private static int sumDigitsMath(int number) {
    return String.valueOf(Math.abs(number)).chars().map(c -> c - '0').sum();
}
```

## 10. Sort List of Strings Alphabetically

```java
import java.util.Arrays;
import java.util.Comparator;
import java.util.List;
import java.util.stream.Collectors;

public class SortStrings {
    public static void main(String[] args) {
        List<String> fruits = Arrays.asList("Banana", "Apple", "Mango", "Orange", "Grape", "apple");

        // Natural lexicographic order (case-sensitive: uppercase before lowercase)
        List<String> sortedCaseSensitive = fruits.stream().sorted().collect(Collectors.toList());

        // Built-in case-insensitive comparator
        List<String> sortedCaseInsensitive = fruits.stream()
                .sorted(String.CASE_INSENSITIVE_ORDER)
                .collect(Collectors.toList());

        // Primary: length; secondary: alphabetical (case-insensitive)
        List<String> sortedByLengthThenAlpha = fruits.stream()
                .sorted(Comparator.comparing(String::length)
                        .thenComparing(String::compareToIgnoreCase))
                .collect(Collectors.toList());
    }
}
```

## 12. Find First Repeated Character in a String

```java
import java.util.HashSet;
import java.util.Optional;
import java.util.Set;

public class FirstRepeatedCharacter {
    public static void main(String[] args) {
        String input = "programming";

        Optional<Character> firstRepeated = findFirstRepeatedChar(input);

        firstRepeated.ifPresentOrElse(
            ch -> System.out.println("First repeated character: " + ch),
            () -> System.out.println("No repeated characters found.")
        );
    }

    private static Optional<Character> findFirstRepeatedChar(String str) {
        Set<Character> seen = new HashSet<>();
        // add() returns false if already present → that char is a repeat
        return str.chars()
                 .mapToObj(c -> (char) c)
                 .filter(c -> !seen.add(c))
                 .findFirst();
    }
}
```

## 15. Find the Minimum and Maximum Number of a Stream

```java
import java.util.Arrays;
import java.util.List;
import java.util.Optional;
import java.util.OptionalInt;
import java.util.stream.IntStream;

public class MinMaxInStream {
    public static void main(String[] args) {
        // Primitive stream: min()/max() return OptionalInt
        OptionalInt min = IntStream.of(5, 3, 9, 1, 7, 4, 6, 8, 2).min();
        OptionalInt max = IntStream.of(5, 3, 9, 1, 7, 4, 6, 8, 2).max();  // recreate; stream already consumed

        List<Integer> numberList = Arrays.asList(5, 3, 9, 1, 7, 4, 6, 8, 2);

        // Object stream requires a Comparator
        Optional<Integer> minObj = numberList.stream().min(Integer::compare);
        Optional<Integer> maxObj = numberList.stream().max(Integer::compare);

        // Single-pass: reduce accumulates [min, max] array simultaneously
        int[] result = numberList.stream()
                .reduce(
                    new int[]{Integer.MAX_VALUE, Integer.MIN_VALUE},
                    (acc, num) -> new int[]{ Math.min(acc[0], num), Math.max(acc[1], num) },
                    (acc1, acc2) -> new int[]{ Math.min(acc1[0], acc2[0]), Math.max(acc1[1], acc2[1]) }
                );

        System.out.println("Min: " + result[0] + ", Max: " + result[1]);
    }
}
```

## 16. Find Prime Number

```java
import java.util.List;
import java.util.stream.Collectors;
import java.util.stream.IntStream;

public class PrimeNumbers {
    public static void main(String[] args) {
        System.out.println(17 + " is prime: " + isPrime(17));
        System.out.println("Primes up to 50: " + findPrimesUpTo(50));
        System.out.println("First 10 primes: " + findFirstNPrimes(10));
    }

    // noneMatch checks no factor exists in [2, sqrt(n)]
    private static boolean isPrime(int number) {
        if (number <= 1) return false;
        return IntStream.rangeClosed(2, (int) Math.sqrt(number))
                .noneMatch(n -> number % n == 0);
    }

    private static List<Integer> findPrimesUpTo(int limit) {
        return IntStream.rangeClosed(2, limit)
                .filter(PrimeNumbers::isPrime)
                .boxed()
                .collect(Collectors.toList());
    }

    // iterate() produces infinite stream; limit() stops at n primes
    private static List<Integer> findFirstNPrimes(int n) {
        return IntStream.iterate(2, i -> i + 1)
                .filter(PrimeNumbers::isPrime)
                .limit(n)
                .boxed()
                .collect(Collectors.toList());
    }
}
```

## 17. Group Elements by a Property

```java
import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

public class GroupingElements {
    public static void main(String[] args) {
        List<Person> people = Arrays.asList(
            new Person("Alice", 30, "New York"),
            new Person("Bob", 25, "London"),
            new Person("Charlie", 30, "New York"),
            new Person("David", 25, "Paris"),
            new Person("Eve", 35, "London")
        );

        // groupingBy produces Map<Key, List<Value>>
        Map<String, List<Person>> groupedByCity = people.stream()
                .collect(Collectors.groupingBy(Person::getCity));

                    else return "Senior";

        // Downstream collector counting() replaces List with count
        Map<String, Long> countByCity = people.stream()
                .collect(Collectors.groupingBy(Person::getCity, Collectors.counting()));
    }
}
```

```java
import java.util.List;
import java.util.Map;
import java.util.function.Function;
import java.util.stream.Collectors;

public class ListToMap {
    public static void main(String[] args) {
        List<Employee> employees = Arrays.asList(
            new Employee(1, "Alice", 50000),
            new Employee(2, "Bob", 60000),
            new Employee(3, "Charlie", 55000),
            new Employee(4, "David", 65000)
        );

        // toMap(keyMapper, valueMapper); Function.identity() keeps the element as value
        Map<Integer, Employee> employeeMap = employees.stream()
                .collect(Collectors.toMap(Employee::getId, Function.identity()));

        Map<String, Double> nameSalaryMap = employees.stream()
                .collect(Collectors.toMap(Employee::getName, Employee::getSalary));

        // Third param is merge function for duplicate keys; Integer::sum accumulates counts
        List<String> words = Arrays.asList("apple", "banana", "apple", "cherry", "banana");
        Map<String, Integer> wordCount = words.stream()
                .collect(Collectors.toMap(Function.identity(), w -> 1, Integer::sum));

        // toConcurrentMap for thread-safe maps in parallel streams
        Map<Integer, String> concurrentMap = employees.parallelStream()
                .collect(Collectors.toConcurrentMap(Employee::getId, Employee::getName));
    }
}
```

## 19. Parallel Processing with Streams

```java
import java.util.List;
import java.util.stream.IntStream;

public class ParallelStreams {
    public static void main(String[] args) {
        List<Integer> largeList = IntStream.range(1, 10_000_000)
                .boxed()
                .collect(java.util.stream.Collectors.toList());

        // Sequential baseline
        long startTime = System.currentTimeMillis();
        long sequentialSum = largeList.stream().mapToLong(Integer::longValue).sum();
        long sequentialTime = System.currentTimeMillis() - startTime;

        // parallelStream() splits work across CPU cores via ForkJoinPool
        startTime = System.currentTimeMillis();
        long parallelSum = largeList.parallelStream().mapToLong(Integer::longValue).sum();
        long parallelTime = System.currentTimeMillis() - startTime;

        System.out.println("Speedup: " + (double) sequentialTime / parallelTime + "x");

        // Parallel is NOT worth it for small datasets due to thread overhead
        List<Integer> smallList = Arrays.asList(1, 2, 3, 4, 5);
        int result = smallList.parallelStream().mapToInt(Integer::intValue).sum();
    }
}
```

## 20. Working with Optional

```java
import java.util.Arrays;
import java.util.List;
import java.util.Optional;
import java.util.stream.Stream;

public class OptionalExample {
    public static void main(String[] args) {
        Optional<String> empty    = Optional.empty();
        Optional<String> nonEmpty = Optional.of("Hello");

        // ofNullable handles null without throwing
        Optional<String> nullable = Optional.ofNullable(null);

        nonEmpty.ifPresent(v -> System.out.println("Present: " + v));

        // ifPresentOrElse handles both branches
        nullable.ifPresentOrElse(
            v -> System.out.println("Value: " + v),
            () -> System.out.println("No value")
        );

        // orElse always evaluates argument; orElseGet is lazy (Supplier)
        String fallback     = empty.orElse("Fallback");
        String lazyFallback = empty.orElseGet(() -> "Computed Fallback");

        // orElseThrow throws if absent
        try { empty.orElseThrow(() -> new RuntimeException("Missing")); }
        catch (Exception e) { System.out.println(e.getMessage()); }

        // map transforms value if present; returns empty Optional otherwise
        String upper = nonEmpty.map(String::toUpperCase).orElse("none");

        // filter returns empty if predicate fails
        Optional<String> filtered = nonEmpty.filter(v -> v.length() > 10);

        // flatMap unwraps nested Optionals — avoids Optional<Optional<T>>
        String userCountry = findUserById(1)
            .flatMap(User::getAddress)
            .flatMap(Address::getCountry)
            .map(Country::getName)
            .orElse("Unknown");

        // stream() converts Optional to 0- or 1-element Stream
        List<String> values = Stream.of(Optional.empty(), Optional.of("Hello"),
                                        Optional.empty(), Optional.of("World"))
            .filter(Optional::isPresent)
            .map(Optional::get)
            .toList();
    }

    // --- Supporting classes ---
    static class User {
        private final Optional<Address> address;
        User(Optional<Address> address) { this.address = address; }
        public Optional<Address> getAddress() { return address; }
    }

    static class Address {
        private final Optional<Country> country;
        Address(Optional<Country> country) { this.country = country; }
        public Optional<Country> getCountry() { return country; }
    }

    static class Country {
        private final String name;
        Country(String name) { this.name = name; }
        public String getName() { return name; }
    }

    private static Optional<User> findUserById(int id) {
        if (id == 1)
            return Optional.of(new User(Optional.of(new Address(Optional.of(new Country("USA"))))));
        return Optional.empty();
    }
}
```
